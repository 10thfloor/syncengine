# Syncengine DevTools Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a Meteor-style floating devtools popover that auto-injects in dev mode via the vite plugin, showing stack health, sync state, data inspection, live message log, and quick actions.

**Architecture:** A new `devtoolsPlugin` sub-plugin in `packages/vite-plugin` injects a self-contained vanilla JS script via `transformIndexHtml`. The script renders into a shadow DOM root. Server-side metrics come from a middleware endpoint; client-side state comes from the data worker via `BroadcastChannel`.

**Tech Stack:** Vanilla JS (shadow DOM), CSS variables, Vite plugin API (`transformIndexHtml`, `configureServer`), `BroadcastChannel`, `fetch` for server metrics/actions.

**Spec:** `docs/superpowers/specs/2026-04-14-devtools-design.md`

- [ ] **Step 1: Add devtools channel setup at module scope**

After the existing `topicState` block (~line 182), add:

```javascript
// -- DevTools BroadcastChannel --

let devtoolsChannel = null;
let devtoolsActive = false;
let devtoolsHeartbeat = null;

function initDevtoolsChannel() {
    if (devtoolsChannel) return;
    try {
        devtoolsChannel = new BroadcastChannel('syncengine-devtools');
        devtoolsChannel.onmessage = (e) => {
            if (e.data && e.data.type === 'devtools-ping') {
                devtoolsActive = true;
                broadcastDevtoolsStatus();
                if (!devtoolsHeartbeat) {
                    devtoolsHeartbeat = setInterval(broadcastDevtoolsStatus, 2000);
                }
            } else if (e.data && e.data.type === 'devtools-action') {
                handleDevtoolsAction(e.data.action);
            }
        };
    } catch {
        // BroadcastChannel not supported
    }
}

function handleDevtoolsAction(action) {
    if (action === 'force-reconnect') {
        if (nats.gwWs) { nats.gwWs.close(); nats.gwWs = null; }
        if (nats.conn) { nats.conn.close(); nats.conn = null; }
    }
    if (action === 'clear-client-db') {
        self.postMessage({ type: 'DEVTOOLS_CLEAR_DB' });
    }
}

function broadcastDevtoolsStatus() {
    if (!devtoolsChannel || !devtoolsActive) return;
    try {
        devtoolsChannel.postMessage({
            type: 'devtools-status',
            sync: {
                phase: sync.phase,
                messagesReplayed: replayCoord.totalProcessed || 0,
                totalMessages: replayCoord.totalExpected || 0,
                snapshotLoaded: !!snapshotState.loaded,
            },
            connection: connectionStatus,
            hlc: { ts: hlcTs, counter: hlcCount },
            conflicts: conflictLog.filter(c => !c.dismissed),
            offlineQueue: causalQueue.length,
            undoDepth: undoStack.length,
            schema: {
                version: schemaState.version || 0,
                fingerprint: schemaState.fingerprint || '',
            },
            entities: [],
            channels: nats.routing ? nats.routing.subjects.map(s => {
                const parts = s.split('.');
                return parts[parts.length - 2] || s;
            }) : [],
            views: Object.assign({}, viewRowCounts),
        });
    } catch { /* channel closed */ }
}

function broadcastDevtoolsMessage(kind, detail) {
    if (!devtoolsChannel || !devtoolsActive) return;
    try {
        devtoolsChannel.postMessage({
            type: 'devtools-message',
            ts: Date.now(),
            kind,
            ...detail,
        });
    } catch { /* channel closed */ }
}
```

```typescript
// packages/vite-plugin/src/devtools/devtools-plugin.ts

import { existsSync, readFileSync } from 'node:fs';
import { resolve } from 'node:path';
import { fileURLToPath } from 'node:url';
import type { Connect, Plugin } from 'vite';

const __dirname = fileURLToPath(new URL('.', import.meta.url));

export function devtoolsPlugin(): Plugin {
    let isDev = false;
    let viteRoot = '';

    return {
        name: 'syncengine:devtools',

        configResolved(config) {
            isDev = config.command === 'serve';
            viteRoot = config.root;
        },

        configureServer(server) {
            if (!isDev) return;
            server.middlewares.use(devtoolsMiddleware(() => readRuntime(viteRoot)));
        },

        transformIndexHtml(html) {
            if (!isDev) return html;

            const clientJs = readFileSync(resolve(__dirname, 'devtools-client.js'), 'utf8');
            const stylesCss = readFileSync(resolve(__dirname, 'devtools-styles.css'), 'utf8');

            const injection = `<script type="module">
const __DEVTOOLS_STYLES__ = ${JSON.stringify(stylesCss)};
${clientJs}
</script>`;

            return html.replace('</body>', injection + '\n</body>');
        },
    };
}

// -- Runtime config reader --

interface DevRuntime {
    natsUrl?: string;
    restateUrl?: string;
    gatewayUrl?: string;
}

function readRuntime(root: string): DevRuntime {
    const p = resolve(root, '.syncengine', 'dev', 'runtime.json');
    if (!existsSync(p)) return {};
    try { return JSON.parse(readFileSync(p, 'utf8')); } catch { return {}; }
}

// -- Metrics & Action middleware --

function devtoolsMiddleware(runtimeFn: () => DevRuntime): Connect.NextHandleFunction {
    return async (req, res, next) => {
        if (!req.url?.startsWith('/__syncengine/devtools/')) return next();

        const pathname = req.url.split('?')[0]!;

        if (pathname === '/__syncengine/devtools/metrics' && req.method === 'GET') {
            return handleMetrics(runtimeFn(), res);
        }
        if (pathname === '/__syncengine/devtools/action' && req.method === 'POST') {
            const chunks: Buffer[] = [];
            for await (const chunk of req) chunks.push(chunk as Buffer);
            const body = JSON.parse(Buffer.concat(chunks).toString('utf8') || '{}');
            return handleAction(runtimeFn(), body, res);
        }

        res.statusCode = 404;
        res.end('Not Found');
    };
}

async function handleMetrics(runtime: DevRuntime, res: Connect.ServerResponse): Promise<void> {
    const natsMonitor = 'http://127.0.0.1:8222';
    const restateAdmin = runtime.restateUrl?.replace(':8080', ':9070') || 'http://127.0.0.1:9070';
    const restateIngress = runtime.restateUrl || 'http://127.0.0.1:8080';

    const result: Record<string, unknown> = { nats: null, restate: null, workspace: null };

    // NATS JetStream stats
    try {
        const r = await fetch(natsMonitor + '/jsz?streams=true');
        const jsz = await r.json() as any;
        const streams = (jsz.account_details ?? [])
            .flatMap((a: any) => a.stream_detail ?? [])
            .filter((s: any) => s.name?.startsWith('WS_'))
            .map((s: any) => ({
                name: s.name,
                messages: s.state?.messages ?? 0,
                bytes: s.state?.bytes ?? 0,
                firstSeq: s.state?.first_seq ?? 0,
                lastSeq: s.state?.last_seq ?? 0,
                consumerCount: s.state?.consumer_count ?? 0,
            }));
        result.nats = { streams };
    } catch { result.nats = { error: 'unreachable' }; }

    // Restate health
    try {
        const r = await fetch(restateAdmin + '/health');
        const health = await r.json() as any;
        result.restate = { healthy: true, services: health.services ?? [] };
    } catch { result.restate = { healthy: false, services: [] }; }

    // Workspace state
    const wsStream = (result.nats as any)?.streams?.[0];
    if (wsStream) {
        const wsId = wsStream.name.replace('WS_', '').replace(/_/g, '-');
        try {
            const stateR = await fetch(restateIngress + '/workspace/' + wsId + '/getState', {
                method: 'POST',
                headers: { 'content-type': 'application/json' },
                body: '[]',
            });
            const state = await stateR.json();
            const membersR = await fetch(restateIngress + '/workspace/' + wsId + '/listMembers', {
                method: 'POST',
                headers: { 'content-type': 'application/json' },
                body: '[]',
            });
            const members = await membersR.json();
            result.workspace = {
                id: wsId,
                active: state?.workspaceId != null,
                members: Array.isArray(members) ? members : [],
                schemaVersion: state?.schemaVersion ?? 0,
            };
        } catch {
            result.workspace = { id: wsId, active: false, members: [], schemaVersion: 0 };
        }
    }

    res.statusCode = 200;
    res.setHeader('content-type', 'application/json');
    res.end(JSON.stringify(result));
}

async function handleAction(
    runtime: DevRuntime,
    body: { action: string },
    res: Connect.ServerResponse,
): Promise<void> {
    const restateIngress = runtime.restateUrl || 'http://127.0.0.1:8080';
    const natsMonitor = 'http://127.0.0.1:8222';

    const ok = (msg: string) => {
        res.statusCode = 200;
        res.setHeader('content-type', 'application/json');
        res.end(JSON.stringify({ ok: true, message: msg }));
    };
    const fail = (msg: string) => {
        res.statusCode = 500;
        res.setHeader('content-type', 'application/json');
        res.end(JSON.stringify({ ok: false, error: msg }));
    };

    // Resolve workspace ID from NATS streams
    let wsId: string | null = null;
    try {
        const r = await fetch(natsMonitor + '/jsz?streams=true');
        const jsz = await r.json() as any;
        const stream = (jsz.account_details ?? [])
            .flatMap((a: any) => a.stream_detail ?? [])
            .find((s: any) => s.name?.startsWith('WS_'));
        if (stream) wsId = stream.name.replace('WS_', '').replace(/_/g, '-');
    } catch { /* */ }

    try {
        switch (body.action) {
            case 'force-reconnect':
            case 'clear-client-db':
                return ok('client-only');

            case 'purge-stream': {
                if (!wsId) return fail('No workspace stream found');
                const streamName = 'WS_' + wsId.replace(/-/g, '_');
                const { connect } = await import('@nats-io/transport-node');
                const nc = await connect({ servers: 'nats://127.0.0.1:4222' });
                const { jetstreamManager } = await import('@nats-io/jetstream');
                const jsm = await jetstreamManager(nc);
                const info = await jsm.streams.purge(streamName);
                await nc.close();
                return ok('Purged ' + info.purged + ' messages');
            }

            case 'trigger-gc': {
                if (!wsId) return fail('No workspace found');
                const r = await fetch(restateIngress + '/workspace/' + wsId + '/triggerGC', {
                    method: 'POST',
                    headers: { 'content-type': 'application/json' },
                    body: '[]',
                });
                const gcResult = await r.json();
                return ok('GC complete: ' + JSON.stringify(gcResult));
            }

            case 'teardown': {
                if (!wsId) return fail('No workspace found');
                await fetch(restateIngress + '/workspace/' + wsId + '/teardown', {
                    method: 'POST',
                    headers: { 'content-type': 'application/json' },
                    body: '[]',
                });
                return ok('Workspace torn down');
            }

            case 'reset': {
                if (!wsId) return fail('No workspace found');
                try {
                    const streamName = 'WS_' + wsId.replace(/-/g, '_');
                    const { connect } = await import('@nats-io/transport-node');
                    const nc = await connect({ servers: 'nats://127.0.0.1:4222' });
                    const { jetstreamManager } = await import('@nats-io/jetstream');
                    const jsm = await jetstreamManager(nc);
                    await jsm.streams.purge(streamName);
                    await nc.close();
                } catch { /* stream may not exist */ }
                await fetch(restateIngress + '/workspace/' + wsId + '/teardown', {
                    method: 'POST',
                    headers: { 'content-type': 'application/json' },
                    body: '[]',
                });
                await fetch(restateIngress + '/workspace/' + wsId + '/provision', {
                    method: 'POST',
                    headers: { 'content-type': 'application/json' },
                    body: JSON.stringify([{ workspaceId: wsId, tenantId: 'default' }]),
                });
                return ok('Full reset complete - reload the page');
            }

            default:
                return fail('Unknown action: ' + body.action);
        }
    } catch (err) {
        return fail((err as Error).message);
    }
}
```

async function execAction(action, confirmMsg) {
    if (confirmMsg && !confirm(confirmMsg)) return;

    // Client-side actions via BroadcastChannel
    if (action === 'force-reconnect' || action === 'clear-client-db') {
        channel?.postMessage({ type: 'devtools-action', action });
        showToast(action === 'force-reconnect' ? 'Reconnecting...' : 'Client DB cleared, re-syncing...');
        return;
    }

    // Server-side actions
    try {
        const r = await fetch('/__syncengine/devtools/action', {
            method: 'POST',
            headers: { 'content-type': 'application/json' },
            body: JSON.stringify({ action }),
        });
        const result = await r.json();
        showToast(result.ok ? result.message : result.error, !result.ok);
    } catch (err) {
        showToast(err.message, true);
    }
}
