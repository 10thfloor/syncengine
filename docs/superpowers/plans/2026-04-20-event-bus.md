# Event Bus Phase 1 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship `bus()` as the third effect type in the hex framework, a unified `publish()` verb for both declarative (entity effect) and imperative (workflow / webhook / heartbeat) use, and deprecate `trigger()` with a one-release migration path. Phase 1 scope is the primitive, the subscriber wiring, defaults + auto-DLQ, CLI scaffolding, and an integration smoke on `apps/notepad`.

**Architecture:** New `bus` primitive in `@syncengine/core` alongside a minimal factory namespace for Layer 1+2 config (`Retention`, `Delivery`, `Storage`, `Retry`, `Backoff`). New `publish()` effect in `emit({ effects })`. `defineWorkflow` accepts `on:` for subscribers and reuses the existing services-injection path. A dispatcher in `@syncengine/server` opens one durable JetStream consumer per subscriber and routes messages to Restate workflow invocations. `trigger()` stays functional but logs a deprecation warning with a migration pointer.

**Tech Stack:** TypeScript, Vitest, `@nats-io/jetstream`, `@nats-io/transport-node`, `@syncengine/gateway-core`'s existing NATS connect helpers, existing syncengine DSL patterns.

**Spec:** `docs/superpowers/specs/2026-04-20-event-bus-design.md`

**Out of scope for Phase 1** (land in Phase 2+):

- Layer 3 `JetStream.*` escape hatch
- `.orderedBy`, `.ordered`, `.concurrency`, `.rate`, `.key` modifiers
- `BusMode.inMemory()` for tests (tests in Phase 1 use a real compose stack)
- Devtools Buses tab
- Scale-out cross-tier smoke (covered by Phase 3 against `docker-compose.serve.yml`)

```ts
// packages/core/src/duration.ts
const DURATION_BRAND: unique symbol = Symbol.for('syncengine.duration');

export interface Duration {
    readonly [DURATION_BRAND]: true;
    readonly ms: number;
}

function build(ms: number, source: string): Duration {
    if (!Number.isInteger(ms)) {
        throw new Error(`${source}: count must be an integer`);
    }
    if (ms < 0) {
        throw new Error(`${source}: count must be non-negative`);
    }
    return { [DURATION_BRAND]: true, ms } as Duration;
}

export const milliseconds = (n: number): Duration => build(n, 'milliseconds');
export const seconds = (n: number): Duration => build(n * 1000, 'seconds');
export const minutes = (n: number): Duration => build(n * 60_000, 'minutes');
export const hours = (n: number): Duration => build(n * 3_600_000, 'hours');
export const days = (n: number): Duration => build(n * 86_400_000, 'days');
```

```ts
// packages/core/src/bus-config.ts
import type { Duration } from './duration';

export type RetentionConfig = {
    readonly kind: 'limits';
    readonly maxAge: Duration;
    readonly maxMsgs?: number;
    readonly discard: 'old' | 'new';
    maxMessages(n: number): RetentionConfig;
    discardOldest(): RetentionConfig;
    discardNewest(): RetentionConfig;
};

function buildRetention(maxAge: Duration, overrides: Partial<RetentionConfig> = {}): RetentionConfig {
    const base: Omit<RetentionConfig, 'maxMessages' | 'discardOldest' | 'discardNewest'> = {
        kind: 'limits',
        maxAge,
        discard: overrides.discard ?? 'old',
        ...(overrides.maxMsgs !== undefined ? { maxMsgs: overrides.maxMsgs } : {}),
    };
    return {
        ...base,
        maxMessages: (n) => buildRetention(maxAge, { ...overrides, maxMsgs: n }),
        discardOldest: () => buildRetention(maxAge, { ...overrides, discard: 'old' }),
        discardNewest: () => buildRetention(maxAge, { ...overrides, discard: 'new' }),
    };
}

export const Retention = {
    durableFor: (maxAge: Duration) => buildRetention(maxAge),
};

export type DeliveryConfig = { readonly mode: 'fanout' | 'queue' | 'interest' };
export const Delivery = {
    fanout: (): DeliveryConfig => ({ mode: 'fanout' }),
    queue: (): DeliveryConfig => ({ mode: 'queue' }),
    interest: (): DeliveryConfig => ({ mode: 'interest' }),
};

export type StorageConfig =
    | { readonly kind: 'file'; readonly replicas: number }
    | { readonly kind: 'memory'; readonly replicas: 1 };

export const Storage = {
    file: (): StorageConfig => ({ kind: 'file', replicas: 1 }),
    memory: (): StorageConfig => ({ kind: 'memory', replicas: 1 }),
    replicatedFile: (opts: { replicas: number }): StorageConfig => {
        if (opts.replicas < 1) {
            throw new Error('Storage.replicatedFile: replicas must be >= 1');
        }
        return { kind: 'file', replicas: opts.replicas };
    },
};

export type BackoffConfig =
    | { readonly kind: 'exponential'; readonly initial: Duration; readonly max: Duration }
    | { readonly kind: 'fixed'; readonly interval: Duration };

export const Backoff = {
    exponential: (opts: { initial: Duration; max: Duration }): BackoffConfig => ({
        kind: 'exponential', initial: opts.initial, max: opts.max,
    }),
    fixed: (opts: { interval: Duration }): BackoffConfig => ({
        kind: 'fixed', interval: opts.interval,
    }),
};

export type RetryConfig =
    | { readonly kind: 'exponential'; readonly attempts: number; readonly initial: Duration; readonly max: Duration }
    | { readonly kind: 'fixed'; readonly attempts: number; readonly interval: Duration }
    | { readonly kind: 'none' };

export const Retry = {
    exponential: (opts: { attempts: number; initial: Duration; max: Duration }): RetryConfig => ({
        kind: 'exponential', ...opts,
    }),
    fixed: (opts: { attempts: number; interval: Duration }): RetryConfig => ({
        kind: 'fixed', ...opts,
    }),
    none: (): RetryConfig => ({ kind: 'none' }),
};
```

```ts
// packages/core/src/bus.ts
import { z, type ZodType } from 'zod';
import type { Duration } from './duration';
import { minutes, days } from './duration';
import {
    Retention, Delivery, Storage,
    type RetentionConfig, type DeliveryConfig, type StorageConfig,
} from './bus-config';

const BUS_NAME_REGEX = /^[a-zA-Z][a-zA-Z0-9_.-]*$/;

export interface DeadEvent<T> {
    readonly original: T;
    readonly error: { message: string; code?: string; stack?: string };
    readonly attempts: number;
    readonly firstAttemptAt: number;
    readonly lastAttemptAt: number;
    readonly workflow: string;
}

export const deadEventSchema = <T>(inner: ZodType<T>) => z.object({
    original: inner,
    error: z.object({
        message: z.string(),
        code: z.string().optional(),
        stack: z.string().optional(),
    }),
    attempts: z.number(),
    firstAttemptAt: z.number(),
    lastAttemptAt: z.number(),
    workflow: z.string(),
});

export interface BusConfig {
    readonly retention: RetentionConfig;
    readonly delivery: DeliveryConfig;
    readonly storage: StorageConfig;
    readonly dedupWindow: Duration;
}

export interface BusRef<T> {
    readonly $tag: 'bus';
    readonly $name: string;
    readonly $schema: ZodType<T>;
    readonly $config: BusConfig;
    /** Auto-generated DLQ bus sharing this bus's lifecycle. */
    readonly dlq: BusRef<DeadEvent<T>>;
}

export interface BusOptions<T> {
    readonly schema: ZodType<T>;
    readonly retention?: RetentionConfig;
    readonly delivery?: DeliveryConfig;
    readonly storage?: StorageConfig;
    readonly dedupWindow?: Duration;
}

function applyDefaults(opts: BusOptions<unknown>): BusConfig {
    return {
        retention: opts.retention ?? Retention.durableFor(days(7)).maxMessages(1_000_000),
        delivery: opts.delivery ?? Delivery.fanout(),
        storage: opts.storage ?? Storage.file(),
        dedupWindow: opts.dedupWindow ?? minutes(1),
    };
}

function buildDlqRef<T>(parentName: string, parentSchema: ZodType<T>): BusRef<DeadEvent<T>> {
    const dlqName = `${parentName}.dlq`;
    const dlqSchema = deadEventSchema(parentSchema);
    const dlqConfig: BusConfig = applyDefaults({
        schema: dlqSchema,
        retention: Retention.durableFor(days(30)),
    });
    // DLQ's dlq is itself (prevents infinite recursion without breaking API).
    const dlq: BusRef<DeadEvent<T>> = {
        $tag: 'bus',
        $name: dlqName,
        $schema: dlqSchema,
        $config: dlqConfig,
        get dlq() { return dlq; },
    };
    return dlq;
}

export function bus<T>(name: string, opts: BusOptions<T>): BusRef<T> {
    if (!name || typeof name !== 'string') {
        throw new Error(`bus(): name must be a non-empty string`);
    }
    if (name.startsWith('$') || name.startsWith('_')) {
        throw new Error(`bus('${name}'): names starting with $ or _ are reserved`);
    }
    if (!BUS_NAME_REGEX.test(name)) {
        throw new Error(`bus('${name}'): name must match ${BUS_NAME_REGEX}`);
    }

    const config = applyDefaults(opts);
    const dlq = buildDlqRef(name, opts.schema);
    return {
        $tag: 'bus',
        $name: name,
        $schema: opts.schema,
        $config: config,
        dlq,
    };
}

export function isBus(value: unknown): value is BusRef<unknown> {
    return !!value && typeof value === 'object' && (value as { $tag?: unknown }).$tag === 'bus';
}
```

```ts
// Add near existing effect helpers
export const PUBLISH_KEY: unique symbol = Symbol.for("syncengine.publish");

export interface EmitPublish<T = unknown> {
    readonly $effect: 'publish';
    readonly bus: BusRef<T>;
    readonly payload: T;
}

export function publish<T>(busRef: BusRef<T>, payload: T): EmitPublish<T> {
    const parsed = busRef.$schema.safeParse(payload);
    if (!parsed.success) {
        throw new Error(`publish(${busRef.$name}): invalid bus payload — ${parsed.error.message}`);
    }
    return { $effect: 'publish', bus: busRef, payload: parsed.data as T };
}
```

```ts
type Publisher = (ctx: RestateLikeCtx, busName: string, payload: unknown) => Promise<void>;

let activePublisher: Publisher | null = null;

/** Wired by @syncengine/server at boot. Tests can inject a stub. */
export function setBusPublisher(p: Publisher | null): void {
    activePublisher = p;
}

// Attach publish() to each BusRef
function attachPublish<T>(ref: BusRef<T>): BusRef<T> {
    const r = ref as BusRef<T> & { publish: (ctx: RestateLikeCtx, payload: T) => Promise<void> };
    r.publish = async (ctx, payload) => {
        const parsed = r.$schema.safeParse(payload);
        if (!parsed.success) {
            throw new Error(`bus.publish(${r.$name}): invalid bus payload — ${parsed.error.message}`);
        }
        await ctx.run(`bus:${r.$name}:publish`, async () => {
            if (!activePublisher) {
                throw new Error(`bus.publish(${r.$name}): no publisher wired — server bootstrap must call setBusPublisher()`);
            }
            await activePublisher(ctx, r.$name, parsed.data);
        });
    };
    return r;
}
```

```ts
// packages/server/src/bus-on.ts
import type { BusRef } from '@syncengine/core';

export type CursorConfig =
    | { kind: 'beginning' }
    | { kind: 'latest' }
    | { kind: 'sequence'; seq: number }
    | { kind: 'time'; at: string };

export const From = {
    beginning: (): CursorConfig => ({ kind: 'beginning' }),
    latest: (): CursorConfig => ({ kind: 'latest' }),
    sequence: (seq: number): CursorConfig => ({ kind: 'sequence', seq }),
    time: (at: Date | string): CursorConfig => ({
        kind: 'time',
        at: typeof at === 'string' ? at : at.toISOString(),
    }),
};

export interface Subscription<T> {
    readonly $tag: 'bus-subscription';
    readonly bus: BusRef<T>;
    readonly predicate?: (event: T) => boolean;
    readonly cursor?: CursorConfig;
    where(predicate: (event: T) => boolean): Subscription<T>;
    from(cursor: CursorConfig): Subscription<T>;
}

export function on<T>(busRef: BusRef<T>): Subscription<T> {
    const build = (predicate?: Subscription<T>['predicate'], cursor?: CursorConfig): Subscription<T> => ({
        $tag: 'bus-subscription',
        bus: busRef,
        ...(predicate ? { predicate } : {}),
        ...(cursor ? { cursor } : {}),
        where: (p) => build(p, cursor),
        from: (c) => build(predicate, c),
    });
    return build();
}
```

Pseudocode:

```
async start() {
  nc = await connectNats(natsUrl);
  js = jetstream(nc);
  const streamName = `WS_${workspaceId.replace(/-/g, '_')}`;
  const consumerName = `bus:${busName}:${subscriberName}`;
  const consumer = await js.consumers.get(streamName, {
      name: consumerName,
      filter_subjects: [`ws.${workspaceId}.bus.${busName}`],
      deliver_policy: cursorToDeliverPolicy(cursor),
      ack_policy: AckPolicy.Explicit,
      max_deliver: retry.attempts + 1,
      // backoff[] from retry config
  });
  const messages = await consumer.consume();
  for await (const m of messages) {
      if (stopped) break;
      const event = m.json();
      if (filterPredicate && !filterPredicate(event)) { m.ack(); continue; }
      const invocationId = `${busName}:${m.seq}`;
      try {
          await postToRestate(restateUrl, workspaceId, subscriberName, invocationId, event);
          m.ack();
      } catch (err) {
          if (m.info.redeliveryCount >= retry.attempts) {
              await publishDlq(nc, workspaceId, dlqBusName, event, err, m);
              m.ack();
          } else {
              m.nak();  // with backoff header per retry config
          }
      }
  }
}
```
